from data_utils import load_captioned_dataset
from peft import LoraConfig
from transformers import TrainingArguments
from train import train_lora_diffusion
import torch
import os

# Force environment to avoid automatic mixed precision detection
os.environ["ACCELERATE_MIXED_PRECISION"] = "no"

model_id = "runwayml/stable-diffusion-v1-5"
dataset = load_captioned_dataset("/content/drive/MyDrive/data")

# Reducing lora_alpha to 8 to scale down the weight updates
lora_config = LoraConfig(
    r=8,
    lora_alpha=8,
    target_modules=["to_q", "to_k", "to_v", "to_out.0"],
    lora_dropout=0.05
)

# Reducing learning_rate to 5e-6 for better numerical stability
training_args = TrainingArguments(
    output_dir="./lora_output",
    num_train_epochs=50,
    per_device_train_batch_size=1,
    gradient_accumulation_steps=4,
    learning_rate=5e-6,
    fp16=False,
    bf16=False,
    max_grad_norm=1.0,
    logging_steps=5,
    remove_unused_columns=False,
    dataloader_num_workers=0
)

# Run training with forced full precision and stabilized hyper-parameters
train_lora_diffusion(model_id, dataset, lora_config, training_args)
